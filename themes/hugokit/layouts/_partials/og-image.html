{{- /* Per-page OG card – generated at build, one per page. The home page keeps the designed card in static/og/hugokit-og.png (giant wordmark + tagline). All other pages get this: the same plate (assets/og/ plate.png – background #0D0D0F, the mark in the same geometry as the home page's card, "macOS 26+ · free" at the bottom right), with the page's OWN title set in Hanken Grotesk on top. No hand-made images to maintain: a new docs page gets its card by itself. The title is wrapped here in the template, because images.Text does not wrap on its own – there is no width parameter, only x/y. The ceiling is three lines; a title that cannot fit is clipped with an ellipsis rather than running past the edge. If you want to adjust the card's rhythm, it is the four constants below. Returns the absolute URL as a string. If one of the resources fails (font or plate), an empty string is returned, and head.html falls back to the home page's card – a missing OG image is worse than a generic one. */ -}} {{- $wrap := 26 -}}{{- /* characters per title line */ -}} {{- $size := 60 -}}{{- /* the title's point size */ -}} {{- $lead := 76 -}}{{- /* line spacing in the title */ -}} {{- $top := 244 -}}{{- /* the title's first line, y */ -}} {{- $plate := resources.Get "og/plate.png" -}} {{- $display := resources.Get "fonts/HankenGrotesk-Bold.ttf" -}} {{- $mono := resources.Get "fonts/JetBrainsMono-Medium.ttf" -}} {{- if and $plate $display $mono -}} {{- /* Wrapping: place words on the current line as long as it stays under the ceiling. A single word longer than the ceiling gets its own line – it is still nicer than clipping in the middle of a word. */ -}} {{- $lines := slice -}} {{- $cur := "" -}} {{- range $word := split (.Title | plainify) " " -}} {{- $try := cond (eq $cur "") $word (printf "%s %s" $cur $word) -}} {{- if le (len $try) $wrap -}} {{- $cur = $try -}} {{- else -}} {{- with $cur }}{{ $lines = $lines | append . }}{{ end -}} {{- $cur = $word -}} {{- end -}} {{- end -}} {{- with $cur }}{{ $lines = $lines | append . }}{{ end -}} {{- if gt (len $lines) 3 -}} {{- $lines = (first 3 $lines) -}} {{- $last := printf "%s…" (index $lines 2) -}} {{- $lines = (first 2 $lines) | append $last -}} {{- end -}} {{- $filters := slice -}} {{- /* Kicker: the same group the docs hub and the side navigation use, so the card says where in the site the page belongs. Changelog has no group – the section is shown instead. */ -}} {{- $label := .Params.group | default (.Section | humanize) -}} {{- with $label -}} {{- $filters = $filters | append (images.Text (upper .) (dict "color" "#9B98A3" "size" 20 "x" 64 "y" 186 "font" $mono)) -}} {{- end -}} {{- /* The title block does not hang at the top: a short title is pushed down, so the block's center of gravity lands in the same place whether it fills one or three lines. Half of the saved lines – the rest becomes air below. */ -}} {{- $y := add $top (mul (sub 3 (len $lines)) (div $lead 2)) -}} {{- range $lines -}} {{- $filters = $filters | append (images.Text . (dict "color" "#F2F0F4" "size" $size "x" 62 "y" $y "font" $display)) -}} {{- $y = add $y $lead -}} {{- end -}} {{- /* Bottom left: the page's own address, so the card can be read without clicking. A counterweight to "macOS 26+ · free", which is painted into the plate. */ -}} {{- $filters = $filters | append (images.Text (printf "hugokit.com%s" .RelPermalink) (dict "color" "#5F5C66" "size" 21 "x" 64 "y" 538 "font" $mono)) -}} {{- with $plate.Filter $filters -}} {{- .Permalink -}} {{- end -}} {{- end -}}
