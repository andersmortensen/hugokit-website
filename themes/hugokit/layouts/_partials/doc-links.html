{{- /* Docs links for a feature. Context: a slice of docs paths from _index.md, optionally with #anchor: "/docs/getting-started/#create-a-new-site". The link is a chip, not a text line: document icon + "Docs" as kicker + the page's own title + arrow – so it is clear you are leaving the home page and landing in the technical description. The label is taken from the documentation itself – the section heading when there is an anchor, otherwise the page's LinkTitle – so the text can never drift from the docs. If the page (or the anchor) does not exist, nothing is rendered: a wrong link is worse than no link. */ -}} {{- range . -}} {{- $parts := split . "#" -}} {{- $path := index $parts 0 -}} {{- $anchor := "" -}} {{- if gt (len $parts) 1 }}{{ $anchor = index $parts 1 }}{{ end -}} {{- $p := site.GetPage $path -}} {{- with $p -}} {{- $href := $p.RelPermalink -}} {{- $label := $p.LinkTitle -}} {{- if $anchor -}} {{- $href = printf "%s#%s" $p.RelPermalink $anchor -}} {{- /* .Fragments.Headings has a synthetic root at level 0 – the page's h2s are its children (same pattern as docs-nav). */ -}} {{- $hs := $p.Fragments.Headings -}} {{- if and $hs (eq (index $hs 0).Level 0) }}{{ $hs = (index $hs 0).Headings }}{{ end -}} {{- range $hs }} {{- if eq .ID $anchor }}{{ $label = .Title | plainify | htmlUnescape }}{{ end -}} {{- /* The anchor can be an h3 under an h2 (e.g. "Hugo" in First launch) */ -}} {{- range .Headings }} {{- if eq .ID $anchor }}{{ $label = .Title | plainify | htmlUnescape }}{{ end -}} {{- end -}} {{- end -}} {{- end -}} Docs · {{ $label }} → {{- end -}} {{- end -}}
